##. Def: Compute if number is prime
function main(): void
	num: integer
	flag: boolean = false
	write << "number = "
	read >> num
	write << endln
	if num > 1:
		for (i: integer) in range(2, num):
			if (num % i) == 0:
				flag = true
				break
	write << num 
	if flag:
		write << " is not a prime number"
	else:
		write << " is a prime number"
	write << endln
